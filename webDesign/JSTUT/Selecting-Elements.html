آموزش انتخاب المان‌ها در DOM

یادگیری روش‌های مختلف انتخاب و دسترسی به عناصر HTML با JavaScript

انتخاب المان‌ها در DOM چیست؟

انتخاب المان‌ها (Selecting Elements) اولین قدم در کار با DOM است. برای تغییر، حذف یا اضافه کردن المان‌ها در صفحه، ابتدا باید بتوانیم به آن‌ها دسترسی پیدا کنیم.

چرا انتخاب المان‌ها مهم است؟
  • تعامل با کاربر: پاسخ به کلیک‌ها، هاور و سایر رویدادها
  • تغییر محتوا: به‌روزرسانی متن، HTML و استایل‌ها
  • اعتبارسنجی فرم: بررسی و کنترل inputها
  • ایجاد پویایی: ساخت برنامه‌های تعاملی و داینامیک

روش‌های اصلی انتخاب المان‌ها

در JavaScript روش‌های مختلفی برای انتخاب المان‌ها وجود دارد که هر کدام کاربرد خاص خود را دارند:

getElementById()
انتخاب یک المان بر اساس ID منحصر به فرد
HTMLElement | null
// انتخاب المان با ID
const header = document.getElementById('header');
const title = document.getElementById('page-title');
getElementsByClassName()
انتخاب تمام المان‌های دارای کلاس مشخص
HTMLCollection
// انتخاب المان‌های با کلاس
const boxes = document.getElementsByClassName('box');
const items = document.getElementsByClassName('menu-item');
getElementsByTagName()
انتخاب تمام المان‌های با تگ HTML مشخص
HTMLCollection
// انتخاب المان‌های با تگ
const paragraphs = document.getElementsByTagName('p');
const buttons = document.getElementsByTagName('button');
querySelector()
انتخاب اولین المان منطبق با selector CSS
HTMLElement | null
// انتخاب با selector
const firstItem = document.querySelector('.item');
const mainButton = document.querySelector('#submit-btn');
querySelectorAll()
انتخاب تمام المان‌های منطبق با selector CSS
NodeList
// انتخاب تمام المان‌های منطبق
const allItems = document.querySelectorAll('.item');
const inputs = document.querySelectorAll('input[type="text"]');

مقایسه روش‌های انتخاب

روش نوع خروجی زنده (Live) سرعت کاربرد
getElementById HTMLElement ❌ بسیار سریع انتخاب المان منحصر به فرد
getElementsByClassName HTMLCollection ✅ سریع انتخاب با کلاس
getElementsByTagName HTMLCollection ✅ سریع انتخاب با تگ
querySelector HTMLElement ❌ متوسط انتخاب با selector پیچیده
querySelectorAll NodeList ❌ کندتر انتخاب گروهی با selector
تفاوت HTMLCollection و NodeList

HTMLCollection یک collection زنده است (با تغییر DOM به‌روز می‌شود) اما NodeList static است (فقط در زمان ایجاد snapshot می‌گیرد).

خروجی انتخاب المان‌ها:

یک روش انتخاب را امتحان کنید...

// کدهای قابل اجرا در دمو
function selectById() {
    const element = document.getElementById('main-header');
    // کار با المان انتخاب شده
}

function selectByClass() {
    const elements = document.getElementsByClassName('box');
    // کار با المان‌های انتخاب شده
}

function selectByTag() {
    const elements = document.getElementsByTagName('p');
    // کار با المان‌های انتخاب شده
}

function selectQuery() {
    const element = document.querySelector('.item.active');
    // کار با المان انتخاب شده
}

function selectQueryAll() {
    const elements = document.querySelectorAll('.item');
    // کار با المان‌های انتخاب شده
}

انتخاب‌گرهای پیشرفته CSS

با querySelector و querySelectorAll می‌توانید از قدرت کامل انتخاب‌گرهای CSS استفاده کنید:

انتخاب‌گرهای ترکیبی
// المان با دو کلاس
const element = document.querySelector('.btn.primary');

// فرزند مستقیم
const child = document.querySelector('.parent > .child');

// نوه‌ها
const descendant = document.querySelector('.ancestor .descendant');
پseudo-class ها
// اولین المان
const first = document.querySelector('.item:first-child');

// آخرین المان
const last = document.querySelector('.item:last-child');

// المان‌های زوج
const even = document.querySelectorAll('tr:nth-child(even)');

// هاور شده
const hovered = document.querySelector('.btn:hover');
انتخاب‌گرهای attribute
// inputهای خاص
const emailInputs = document.querySelectorAll('input[type="email"]');

// المان‌های دارای attribute
const withTitle = document.querySelectorAll('[title]');

// مقدار دقیق attribute
const exactMatch = document.querySelector('[data-id="123"]');

// حاوی مقدار
const contains = document.querySelector('[class*="error"]');

بهترین روش‌ها و نکات

نکات مهم در انتخاب المان‌ها
  • همیشه قبل از کار با المان، وجود آن را بررسی کنید (if (element) { ... })
  • برای انتخاب‌های مکرر، المان را در متغیر ذخیره کنید
  • از getElementById برای المان‌های منحصر به فرد استفاده کنید
  • برای انتخاب‌های پیچیده از querySelector استفاده کنید
  • HTMLCollection و NodeList شبیه آرایه هستند اما آرایه نیستند
مثال: تبدیل HTMLCollection به آرایه
// HTMLCollection را به آرایه تبدیل کنید
const boxes = document.getElementsByClassName('box');
const boxesArray = Array.from(boxes);

// حالا می‌توانید از متدهای آرایه استفاده کنید
boxesArray.forEach(box => {
    box.style.backgroundColor = 'blue';
});

// یا با spread operator
const boxesArray2 = [...document.getElementsByClassName('box')];
