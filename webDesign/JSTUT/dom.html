آموزش کامل Document Object Model (DOM)

DOM رابط برنامه‌نویسی است که به JavaScript اجازه می‌دهد با صفحات HTML تعامل داشته باشد

DOM چیست؟

DOM (Document Object Model) یک رابط برنامه‌نویسی برای اسناد HTML و XML است. این مدل، سند را به صورت یک ساختار درختی از اشیاء نمایش می‌دهد.

<html>
  <head>
    <title>عنوان صفحه</title>
  </head>
  <body>
    <h1 id="title">سلام دنیا!</h1>
    <button onclick="changeText()">کلیک کن</button>
  </body>
</html>
نکته مهم:

DOM یک رابط مستقل از زبان است، اما ما در اینجا از JavaScript برای کار با آن استفاده می‌کنیم.

۱. انتخاب المان‌ها

روش‌های مختلف برای انتخاب المان‌ها در DOM:

انتخاب با ID
// انتخاب المان با ID مشخص
const title = document.getElementById("title");
const header = document.querySelector("#header");
انتخاب با کلاس
// انتخاب تمام المان‌های با کلاس مشخص
const items = document.getElementsByClassName("item");
const boxes = document.querySelectorAll(".box");
انتخاب با تگ
// انتخاب تمام المان‌های با تگ مشخص
const paragraphs = document.getElementsByTagName("p");
const divs = document.querySelectorAll("div");

۲. تغییر محتوا و استایل

روش‌های تغییر محتوا و ظاهر المان‌ها:

تغییر متن و HTML
// تغییر متن ساده
element.innerText = "متن جدید";

// تغییر HTML (امنیت کمتر)
element.innerHTML = "<strong>متن ضخیم</strong>";

// تغییر متن (امن‌تر)
element.textContent = "متن جدید";
تغییر استایل
// تغییر مستقیم استایل
element.style.color = "red";
element.style.backgroundColor = "blue";
element.style.fontSize = "20px";

// اضافه کردن کلاس
element.classList.add("active");

// حذف کلاس
element.classList.remove("old-class");
دمو: تغییر محتوا و استایل
این المان برای نمایش تغییرات استایل است

۳. ایجاد و حذف المان‌ها

روش‌های ایجاد المان‌های جدید و حذف المان‌های موجود:

ایجاد المان جدید
// ایجاد المان جدید
const newDiv = document.createElement("div");
newDiv.textContent = "این یک div جدید است";

// اضافه کردن به DOM
parentElement.appendChild(newDiv);
حذف المان
// حذف المان
element.remove();

// یا
parentElement.removeChild(element);
دمو: مدیریت لیست
  • مورد اول
  • مورد دوم

۴. رویدادها (Events)

مدیریت رویدادها در DOM:

اضافه کردن Event Listener
// روش مدرن
button.addEventListener("click", function() {
    alert("دکمه کلیک شد!");
});

// روش قدیمی
button.onclick = function() {
    alert("دکمه کلیک شد!");
};
رویدادهای رایج
// کلیک
element.addEventListener("click", handleClick);

// هاور (ورود موس)
element.addEventListener("mouseenter", handleMouseEnter);

// خروج موس
element.addEventListener("mouseleave", handleMouseLeave);

// فشار دکمه صفحه کلید
document.addEventListener("keydown", handleKeyDown);

// تغییر مقدار input
input.addEventListener("input", handleInput);
دمو: رویدادها
هاور کن
خروجی رویدادها اینجا نمایش داده می‌شود

۵. پیمایش DOM

حرکت بین المان‌ها در درخت DOM:

والد و فرزندان
// والد
const parent = element.parentElement;

// فرزندان
const children = element.children;
const firstChild = element.firstElementChild;
const lastChild = element.lastElementChild;

// خواهر و برادرها
const nextSibling = element.nextElementSibling;
const prevSibling = element.previousElementSibling;
تفاوت مهم:

children فقط المان‌ها را برمی‌گرداند، اما childNodes تمام گره‌ها (شامل متن و کامنت) را برمی‌گرداند.

پروژه عملی: Todo List

بیایید یک برنامه Todo List ساده با استفاده از DOM ایجاد کنیم:

برنامه Todo List
کد کامل Todo List
function addTodo() {
    const input = document.getElementById("todo-input");
    const text = input.value.trim();

    if (text === "") return;

    // ایجاد المان جدید
    const li = document.createElement("li");
    li.innerHTML = `
        ${text}
        
        
    `;

    // اضافه کردن به لیست
    document.getElementById("todo-list").appendChild(li);

    // پاک کردن input
    input.value = "";
}

function removeTodo(button) {
    button.parentElement.remove();
}

function toggleTodo(button) {
    const li = button.parentElement;
    li.classList.toggle("completed");
}
