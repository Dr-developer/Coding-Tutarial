توابع در JavaScript چیست؟

تابع (Function) یک بلوک از کد است که برای انجام یک کار خاص طراحی شده است. توابع به ما کمک می‌کنند کدهای تکراری را کاهش دهیم و برنامه‌هایمان را به بخش‌های کوچک‌تر و مدیریت‌پذیرتر تقسیم کنیم.

چرا از توابع استفاده می‌کنیم؟
  • قابلیت استفاده مجدد: یک بار تابع را می‌نویسیم، چندین بار استفاده می‌کنیم
  • سازمان‌دهی کد: برنامه را به بخش‌های منطقی تقسیم می‌کنیم
  • کاهش خطا: با تست یک تابع، از صحت عملکرد آن در تمام برنامه اطمینان پیدا می‌کنیم
  • خوانایی: کد تمیزتر و قابل فهم‌تری خواهیم داشت

انواع روش‌های تعریف تابع

در JavaScript چهار روش اصلی برای تعریف توابع وجود دارد:

۱. تعریف تابع (Function Declaration)
// تعریف تابع
function greet(name) {
    return `سلام ${name}!`;
}

// فراخوانی تابع
const message = greet("علی");
console.log(message); // خروجی: سلام علی!
۲. عبارت تابع (Function Expression)
// تعریف تابع در یک متغیر
const multiply = function(a, b) {
    return a * b;
};

// فراخوانی تابع
const result = multiply(5, 3);
console.log(result); // خروجی: 15
۳. توابع پیکانی (Arrow Functions)
// روش کوتاه‌نویسی
const square = (x) => x * x;

// روش کامل
const add = (a, b) => {
    const sum = a + b;
    return sum;
};

// فراخوانی
console.log(square(4)); // خروجی: 16
console.log(add(2, 3)); // خروجی: 5
۴. سازنده تابع (Function Constructor)
// روش کمتر رایج
const divide = new Function('a', 'b', 'return a / b');

// فراخوانی
console.log(divide(10, 2)); // خروجی: 5
تفاوت مهم Function Declaration و Expression

توابع تعریفی (Declaration) قبل از اجرای کد در حافظه قرار می‌گیرند (Hoisting)، اما توابع عبارتی (Expression) فقط زمانی در دسترس هستند که به خط تعریفشان برسیم.

پارامترها و آرگومان‌ها

پارامتر (Parameter) متغیرهایی هستند که در تعریف تابع می‌آیند. آرگومان (Argument) مقادیر واقعی هستند که هنگام فراخوانی به تابع پاس داده می‌شوند.

پارامترهای پیش‌فرض
function createUser(name, age = 18, isActive = true) {
    return {
        name: name,
        age: age,
        isActive: isActive
    };
}

// استفاده از مقادیر پیش‌فرض
const user1 = createUser("علی"); // age=18, isActive=true
const user2 = createUser("رضا", 25); // isActive=true
const user3 = createUser("سارا", 30, false);
پارامترهای rest
function sum(...numbers) {
    let total = 0;
    for (let num of numbers) {
        total += num;
    }
    return total;
}

console.log(sum(1, 2, 3)); // خروجی: 6
console.log(sum(1, 2, 3, 4, 5)); // خروجی: 15

بازگشت مقدار (Return)

دستور return مقدار خروجی تابع را مشخص می‌کند. اگر تابعی return نداشته باشد، مقدار undefined برمی‌گرداند.

بازگشت چندین مقدار
function calculate(a, b) {
    const sum = a + b;
    const difference = a - b;
    const product = a * b;
    const quotient = a / b;

    return { sum, difference, product, quotient };
}

const results = calculate(10, 5);
console.log(results.sum); // خروجی: 15
console.log(results.product); // خروجی: 50

مفاهیم پیشرفته

توابع بازگشتی (Recursive)
// محاسبه فاکتوریل
function factorial(n) {
    if (n === 0 || n === 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

console.log(factorial(5)); // خروجی: 120
توابع بازگشتی (Closures)
function createCounter() {
    let count = 0;

    return function() {
        count++;
        return count;
    };
}

const counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3
توابع سازنده (Constructors)
function Person(name, age) {
    this.name = name;
    this.age = age;

    this.introduce = function() {
        return `من ${this.name} هستم و ${this.age} سال دارم`;
    };
}

const person1 = new Person("علی", 20);
console.log(person1.introduce());
